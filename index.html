import serial
from Crypto.Cipher import AES
# Secure communication with embedded device
class SecureDevice:
def __init__(self, port):
self.ser = serial.Serial(port, 115200)
self.key = '32-byte-key-for-AES-256-encryption'
def secure_send(self, data):
# Encrypt and send to device
cipher = AES.new(self.key, AES.MODE_EAX)
ciphertext, tag = cipher.encrypt_and_digest(data)
self.ser.write(ciphertext)
def secure_receive(self):
# Receive and decrypt from device
ciphertext = self.ser.read(1024)
cipher = AES.new(self.key, AES.MODE_EAX)
return cipher.decrypt(ciphertext)
